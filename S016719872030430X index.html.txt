function createStars(){
    for(let i=0;i<120;i++){
        let star=document.createElement("div");
        star.classList.add("star");
        star.style.width=Math.random()*3+"px";
        star.style.height=star.style.width;
        star.style.top=Math.random()*100+"vh";
        star.style.left=Math.random()*100+"vw";
        document.body.appendChild(star);
    }
}

function createShootingStars(){
    setInterval(()=>{
        let shoot=document.createElement("div");
        shoot.classList.add("shooting-star");
        shoot.style.top=Math.random()*50+"vh";
        shoot.style.left=Math.random()*50+"vw";
        document.body.appendChild(shoot);
        setTimeout(()=>shoot.remove(),2000);
    },3000);
}

function startFireworks(){
    let sound=document.getElementById("boomSound");
    sound.play();

    for(let i=0;i<8;i++){
        let x=Math.random()*window.innerWidth;
        let y=Math.random()*window.innerHeight;
        createExplosion(x,y);
    }
}

function createExplosion(x,y){
    for(let i=0;i<30;i++){
        let particle=document.createElement("div");
        particle.classList.add("particle");
        particle.style.background=`hsl(${Math.random()*360},100%,60%)`;
        particle.style.left=x+"px";
        particle.style.top=y+"px";
        let angle=Math.random()*2*Math.PI;
        let distance=Math.random()*150;
        particle.style.transform=`translate(${Math.cos(angle)*distance}px,${Math.sin(angle)*distance}px)`;
        document.body.appendChild(particle);
        setTimeout(()=>particle.remove(),1000);
    }
}
